```go
var timeNow = time.Now

func GetDate() int {
	return timeNow().Day()
}

stubs := gostub.Stub(&timeNow, func() time.Time {
  return time.Date(2015, 6, 1, 0, 0, 0, 0, time.UTC)
})
defer stubs.Reset()
```

```go
// production code:
var osHostname = osHostname

// Test code:
stubs := gostub.StubFunc(&osHostname, "fakehost", nil)
defer stubs.Reset()
```
